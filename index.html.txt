    /***********************

    const $ = (id) => document.getElementById(id);

    function showToast(message, ms = 1800) {
      const el = $("toast");
      el.textContent = message;
      el.classList.add("show");
      window.clearTimeout(showToast._t);
      showToast._t = window.setTimeout(() => el.classList.remove("show"), ms);
    }

    function escapeHTML(str) {
      // Prevent injection in feed rendering
      return String(str)
        .replaceAll("&", "&amp;")
        .replaceAll("<", "&lt;")
        .replaceAll(">", "&gt;")
        .replaceAll('"', "&quot;")
        .replaceAll("'", "&#039;");
    }

    function clamp(n, min, max) { return Math.max(min, Math.min(max, n)); }

    /***********************
     * Map Setup
     ***********************/
    const map = L.map("map", { zoomControl: true }).setView([28.6139, 77.2090], 12);

    L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
      attribution: "© OpenStreetMap"
    }).addTo(map);

    let userMarker = null;
    let userCircle = null;

    let routeLine = null;
    let destMarker = null;

    // Keep last known coords (for routing)
    let userLatLng = null;

    function setUserLocation(latlng, accuracyMeters) {
      userLatLng = latlng;

      if (!userMarker) {
        userMarker = L.marker(latlng).addTo(map).bindPopup("📍 You are here!");
      } else {
        userMarker.setLatLng(latlng);
      }

      if (!userCircle) {
        userCircle = L.circle(latlng, { radius: Math.max(10, accuracyMeters / 2) }).addTo(map);
      } else {
        userCircle.setLatLng(latlng).setRadius(Math.max(10, accuracyMeters / 2));
      }
    }

    function locateUser({ setView = true } = {}) {
      if (!navigator.geolocation) {
        showToast("Geolocation not supported on this device.");
        return;
      }

      map.locate({ setView, maxZoom: 16, enableHighAccuracy: true });
    }

    map.on("locationfound", (e) => {
      setUserLocation(e.latlng, e.accuracy || 30);
      if (userMarker) userMarker.openPopup();
    });

    map.on("locationerror", () => {
      showToast("Could not get your location. Enable location permissions.");
    });

    // Request location once on load (same as your V4 but with feedback)
    locateUser({ setView: true });

      $("scoreValue").textContent = "--";
      $("scoreDetails").textContent = "--";

    }

    /***********************

    const STORAGE_KEY = "pathrater_reports_v1";

    function loadReports() {
      try {
        const raw = localStorage.getItem(STORAGE_KEY);
        const data = raw ? JSON.parse(raw) : [];
        return Array.isArray(data) ? data : [];
      } catch {
        return [];
      }
    }

    function saveReports(reports) {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(reports));
    }

      if (reports.length === 0) {

    /***********************
     * Submit Review (fixed + safer)

    async function fileToDataUrl(file) {
      return new Promise((resolve, reject) => {
        const reader = new FileReader();
        reader.onload = (e) => resolve(e.target.result);
        reader.onerror = reject;
        reader.readAsDataURL(file);
      });
    }

    async function submitReview() {
      const location = $("reviewLocation").value.trim();
      const text = $("reviewText").value.trim();
      const fileInput = $("reviewImage");

      if (!activeTag || !location) {
        showToast("Select a hazard tag and enter a location.");
        return;
      }

      // Optional image
      let imageDataUrl = "";
      const file = fileInput.files && fileInput.files[0] ? fileInput.files[0] : null;
      if (file) {
        // Basic size guard (keeps localStorage from exploding)
        const maxBytes = 900 * 1024; // ~900KB
        if (file.size > maxBytes) {
          showToast("Image is large. Please use a smaller photo.");
          return;
        }
        try {
          imageDataUrl = await fileToDataUrl(file);
        } catch {
          showToast("Could not read the photo. Try again.");
          return;
        }
      }

      const reports = loadReports();
      reports.unshift({
        tag: activeTag,
        location,
        text,
        imageDataUrl,
        createdAt: Date.now()
      });
      saveReports(reports);

      // Reset form
      $("reviewLocation").value = "";
      $("reviewText").value = "";
      document.querySelectorAll("#hazardTags .tag").forEach(tag => tag.classList.remove("selected"));
      activeTag = "";
      fileInput.value = "";

      renderFeed();
      showToast("Report posted.");
      // Move to feed
      const feedNav = document.querySelector('.nav-item[data-tab="feed"]');
      feedNav.click();
    }

    /***********************
     * Geocoding + Routing + Walkability (actual route line!)
     * - Geocoding: Nominatim
     * - Routing: OSRM public demo server
     ***********************/
    async function geocodePlace(query) {
      const url = `https://nominatim.openstreetmap.org/search?format=json&limit=1&q=${encodeURIComponent(query)}`;
      const resp = await fetch(url, { headers: { "Accept": "application/json" } });
      if (!resp.ok) throw new Error("geocode_failed");
      const data = await resp.json();
      if (!data || !data[0]) throw new Error("no_results");
      return { lat: parseFloat(data[0].lat), lon: parseFloat(data[0].lon), display: data[0].display_name };
    }

    async function routeOSRM(fromLatLng, toLatLng) {
      const url =
        `https://router.project-osrm.org/route/v1/foot/` +
        `${fromLatLng.lng},${fromLatLng.lat};${toLatLng.lng},${toLatLng.lat}` +
        `?overview=full&geometries=geojson&steps=false`;

      const resp = await fetch(url, { headers: { "Accept": "application/json" } });
      if (!resp.ok) throw new Error("route_failed");

      const data = await resp.json();
      if (!data.routes || !data.routes[0]) throw new Error("route_missing");

      const r = data.routes[0];
      return {
        distanceM: r.distance,
        durationS: r.duration,
        geometry: r.geometry
      };
    }

    function computeWalkabilityScore(distanceM, durationS) {
      // This is a simple heuristic:
      // - Faster walking time for the same distance implies fewer interruptions.
      // - Longer distances get slightly penalized.
      const speed = distanceM / Math.max(1, durationS); // m/s
      const speedScore = clamp((speed - 0.8) / (1.8 - 0.8), 0, 1) * 65; // typical walk ~1.2–1.6 m/s
      const distancePenalty = clamp(distanceM / 2500, 0, 1) * 20; // penalize >2.5km
      const base = 35;
      const score = clamp(Math.round(base + speedScore - distancePenalty), 1, 100);
      return score;
    }

    function formatMeters(m) {
      if (m >= 1000) return `${(m/1000).toFixed(2)} km`;
      return `${Math.round(m)} m`;
    }
    function formatDuration(sec) {
      const min = Math.round(sec / 60);
      if (min < 60) return `${min} min`;
      const h = Math.floor(min / 60);
      const r = min % 60;
      return `${h} hr ${r} min`;
    }

    async function calculateWalkability() {
      const dest = $("pointB").value.trim();
      if (!dest) {
        showToast("Enter a destination.");
        return;
      }
      if (!userLatLng) {
        showToast("Get your location first.");
        return;
      }

      const btn = $("btnWalkability");
      btn.disabled = true;
      btn.textContent = "Working…";

      try {
        // 1) Geocode destination
        const g = await geocodePlace(dest);
        const toLatLng = L.latLng(g.lat, g.lon);

        // 2) Route
        const routed = await routeOSRM(userLatLng, toLatLng);

        // 3) Draw route
        if (routeLine) map.removeLayer(routeLine);
        if (destMarker) map.removeLayer(destMarker);

        destMarker = L.marker(toLatLng).addTo(map).bindPopup("🎯 Destination").openPopup();

        routeLine = L.geoJSON(routed.geometry, {
          style: { weight: 6, opacity: 0.9 }
        }).addTo(map);

        // Fit bounds
        const bounds = routeLine.getBounds();
        map.fitBounds(bounds, { padding: [20, 20] });

        // 4) Score (heuristic)
        const score = computeWalkabilityScore(routed.distanceM, routed.durationS);

        $("scoreValue").textContent = score;
        $("scoreDetails").textContent = `${formatDuration(routed.durationS)} • ${formatMeters(routed.distanceM)}`;
        $("routeDetails").textContent = `Route to: ${g.display}`;
        $("scoreCard").classList.add("active");

        showToast("Route drawn.");
      } catch (e) {
        showToast("Could not route to that destination. Try a different place name.");
      } finally {
        btn.disabled = false;
        btn.textContent = "Get Walkability Score";
      }
    }

    /***********************
     * Export (quick win for sharing)
     ***********************/
    function exportReports() {
      const data = loadReports();
      const blob = new Blob([JSON.stringify(data, null, 2)], { type: "application/json" });
      const url = URL.createObjectURL(blob);

      const a = document.createElement("a");
      a.href = url;
      a.download = "pathrater-reports.json";
      document.body.appendChild(a);
      a.click();
      a.remove();
      URL.revokeObjectURL(url);

      showToast("Exported reports.");
    }
